svg 基本特性七：定义和重写

定义：defs 和重写 use

          // 定义一些元素
          <defs>
            //分组
            <g id="star">
              <svg width="200" height="200" viewBox="0,0,200,200" style="border:1px solid #cd0000;">
                <points="50,50 30,100 75,65 25,65 70,100" />
              </svg>
            </g id="star">
          </defs>
          // 复用与重写
          <use xlink:href="#star" x="100" y="0" style="fill: #ff0; stroke: #f00; stroke-width: 1px"></use>
